class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int maxSofar=INT_MIN;
        int mx=0;
        for(int i=0;i<nums.size();++i){
            mx=max(nums[i],nums[i]+mx);
            if(mx>maxSofar){
                maxSofar=mx;
            }
        }
        return maxSofar;
    }
};
